cmake_minimum_required(VERSION 3.0.0)
project(doomy VERSION 0.1.0)
set(CMAKE_CXX_STANDARD 11)

include(CTest)
enable_testing()

add_library(shared_graphics SHARED src/shared_graphics.cpp)  # Include the empty source file here

add_executable(doomy src/main.cpp src/glad.cpp src/draw.cpp src/std_graphics.cpp src/geom.cpp src/input.cpp src/shared_graphics.cpp)

add_custom_command(TARGET shared_graphics
                    POST_BUILD
                    COMMAND g++ -shared -fPIC src/shared_graphics.cpp -o shared_graphics.so  # Compile the empty source file
                    COMMENT "Compiling shared_graphics.so...")


include_directories(include)

add_subdirectory(lib/GLFW)
target_link_libraries(doomy glfw)

set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
